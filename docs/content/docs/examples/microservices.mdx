A microservices example with service registry and load balancing.

## Schema

```capnp
@0xea1b2c3d4e5f6789;

interface ServiceRegistry {
  register @0 (name :Text, endpoint :ServiceEndpoint) -> (id :Text);
  unregister @1 (id :Text) -> ();
  lookup @2 (name :Text) -> (endpoints :List(ServiceEndpoint));
  subscribe @3 (name :Text, handler :ServiceWatcher) -> (subscription :Subscription);
}

struct ServiceEndpoint {
  host @0 :Text;
  port @1 :UInt16;
  metadata @2 :List(KeyValue);

  struct KeyValue {
    key @0 :Text;
    value @1 :Text;
  }
}

interface ServiceWatcher {
  onAdded @0 (endpoint :ServiceEndpoint) -> ();
  onRemoved @1 (endpoint :ServiceEndpoint) -> ();
}

interface Subscription {
  cancel @0 () -> ();
}
```

## Registry Server

```csharp
using System.Collections.Concurrent;

public class ServiceRegistryImpl : ServiceRegistry.IServer
{
    private readonly ConcurrentDictionary<string, ServiceInfo> _services = new();
    private readonly ConcurrentDictionary<string, List<ServiceWatcher.IClient>> _watchers = new();
    private int _nextId = 0;

    public Task<string> Register(
        string name,
        ServiceEndpoint endpoint,
        CancellationToken ct)
    {
        var id = Interlocked.Increment(ref _nextId).ToString();
        var info = new ServiceInfo(id, name, endpoint);

        _services[id] = info;

        Console.WriteLine($"Registered: {name} at {endpoint.Host}:{endpoint.Port}");

        // Notify watchers
        _ = NotifyWatchersAsync(name, w => w.OnAdded(endpoint));

        return Task.FromResult(id);
    }

    public async Task Unregister(string id, CancellationToken ct)
    {
        if (_services.TryRemove(id, out var info))
        {
            Console.WriteLine($"Unregistered: {info.Name}");
            await NotifyWatchersAsync(info.Name, w => w.OnRemoved(info.Endpoint));
        }
    }

    public Task<IReadOnlyList<ServiceEndpoint>> Lookup(
        string name,
        CancellationToken ct)
    {
        var endpoints = _services.Values
            .Where(s => s.Name == name)
            .Select(s => s.Endpoint)
            .ToList();

        return Task.FromResult<IReadOnlyList<ServiceEndpoint>>(endpoints);
    }

    public Task<Subscription.IClient> Subscribe(
        string name,
        ServiceWatcher.IClient handler,
        CancellationToken ct)
    {
        var watchers = _watchers.GetOrAdd(name, _ => new List<ServiceWatcher.IClient>());

        lock (watchers)
        {
            watchers.Add(handler);
        }

        var subscription = new SubscriptionImpl(this, name, handler);
        return Task.FromResult<Subscription.IClient>(subscription);
    }

    internal void Unsubscribe(string name, ServiceWatcher.IClient handler)
    {
        if (_watchers.TryGetValue(name, out var watchers))
        {
            lock (watchers)
            {
                watchers.Remove(handler);
            }
        }
    }

    private async Task NotifyWatchersAsync(
        string name,
        Func<ServiceWatcher.IClient, Task> action)
    {
        if (!_watchers.TryGetValue(name, out var watchers))
            return;

        List<ServiceWatcher.IClient> copy;
        lock (watchers)
        {
            copy = watchers.ToList();
        }

        foreach (var watcher in copy)
        {
            try
            {
                await action(watcher);
            }
            catch (RpcException)
            {
                // Watcher disconnected
            }
        }
    }

    record ServiceInfo(string Id, string Name, ServiceEndpoint Endpoint);
}

public class SubscriptionImpl : Subscription.IServer
{
    private readonly ServiceRegistryImpl _registry;
    private readonly string _name;
    private readonly ServiceWatcher.IClient _handler;

    public SubscriptionImpl(
        ServiceRegistryImpl registry,
        string name,
        ServiceWatcher.IClient handler)
    {
        _registry = registry;
        _name = name;
        _handler = handler;
    }

    public Task Cancel(CancellationToken ct)
    {
        _registry.Unsubscribe(_name, _handler);
        return Task.CompletedTask;
    }
}
```

## Service Client with Load Balancing

```csharp
public class ServiceClient<T> : IAsyncDisposable where T : class, IDisposable
{
    private readonly ServiceRegistry.IClient _registry;
    private readonly string _serviceName;
    private readonly List<ServiceConnection> _connections = new();
    private readonly SemaphoreSlim _lock = new(1);
    private Subscription.IClient? _subscription;
    private int _roundRobin = 0;

    public ServiceClient(ServiceRegistry.IClient registry, string serviceName)
    {
        _registry = registry;
        _serviceName = serviceName;
    }

    public async Task InitializeAsync()
    {
        // Get initial endpoints
        var endpoints = await _registry.Lookup(_serviceName);

        foreach (var ep in endpoints)
        {
            await AddConnectionAsync(ep);
        }

        // Subscribe to changes
        _subscription = await _registry.Subscribe(
            _serviceName,
            new ServiceWatcherImpl(this));
    }

    public T GetService()
    {
        if (_connections.Count == 0)
            throw new InvalidOperationException($"No {_serviceName} services available");

        // Round-robin load balancing
        var index = Interlocked.Increment(ref _roundRobin) % _connections.Count;
        var conn = _connections[index];

        return conn.Client.GetMain<T>();
    }

    private async Task AddConnectionAsync(ServiceEndpoint endpoint)
    {
        await _lock.WaitAsync();
        try
        {
            // Check if already connected
            if (_connections.Any(c =>
                c.Host == endpoint.Host && c.Port == endpoint.Port))
                return;

            var client = new TcpRpcClient(endpoint.Host, endpoint.Port);
            await client.WhenConnected;

            _connections.Add(new ServiceConnection(
                endpoint.Host,
                endpoint.Port,
                client));

            Console.WriteLine($"Connected to {_serviceName} at {endpoint.Host}:{endpoint.Port}");
        }
        finally
        {
            _lock.Release();
        }
    }

    private async Task RemoveConnectionAsync(ServiceEndpoint endpoint)
    {
        await _lock.WaitAsync();
        try
        {
            var conn = _connections.FirstOrDefault(c =>
                c.Host == endpoint.Host && c.Port == endpoint.Port);

            if (conn != null)
            {
                _connections.Remove(conn);
                conn.Client.Dispose();
                Console.WriteLine($"Disconnected from {_serviceName} at {endpoint.Host}:{endpoint.Port}");
            }
        }
        finally
        {
            _lock.Release();
        }
    }

    public async ValueTask DisposeAsync()
    {
        if (_subscription != null)
        {
            await _subscription.Cancel();
        }

        foreach (var conn in _connections)
        {
            conn.Client.Dispose();
        }
    }

    record ServiceConnection(string Host, ushort Port, TcpRpcClient Client);

    class ServiceWatcherImpl : ServiceWatcher.IServer
    {
        private readonly ServiceClient<T> _parent;

        public ServiceWatcherImpl(ServiceClient<T> parent) => _parent = parent;

        public Task OnAdded(ServiceEndpoint endpoint, CancellationToken ct)
        {
            return _parent.AddConnectionAsync(endpoint);
        }

        public Task OnRemoved(ServiceEndpoint endpoint, CancellationToken ct)
        {
            return _parent.RemoveConnectionAsync(endpoint);
        }
    }
}
```

## Example Calculator Service

```csharp
// calculator.capnp
// interface Calculator {
//   add @0 (a :Float64, b :Float64) -> (result :Float64);
// }

public class CalculatorService
{
    private readonly ServiceRegistry.IClient _registry;
    private readonly int _port;
    private TcpRpcServer? _server;
    private string? _registrationId;

    public CalculatorService(ServiceRegistry.IClient registry, int port)
    {
        _registry = registry;
        _port = port;
    }

    public async Task StartAsync()
    {
        _server = new TcpRpcServer();
        _server.Main = new CalculatorImpl();
        _server.StartAccepting(IPAddress.Any, _port);

        // Register with service registry
        var endpoint = new ServiceEndpoint
        {
            Host = "localhost",
            Port = (ushort)_port,
            Metadata = new List<ServiceEndpoint.KeyValue>
            {
                new() { Key = "version", Value = "1.0" }
            }
        };

        _registrationId = await _registry.Register("calculator", endpoint);
        Console.WriteLine($"Calculator service started on port {_port}");
    }

    public async Task StopAsync()
    {
        if (_registrationId != null)
        {
            await _registry.Unregister(_registrationId);
        }

        _server?.Dispose();
    }
}
```

## Usage Example

```csharp
// Start registry server
using var registryServer = new TcpRpcServer(IPAddress.Any, 9000);
registryServer.Main = new ServiceRegistryImpl();

// Connect to registry
using var registryClient = new TcpRpcClient("localhost", 9000);
await registryClient.WhenConnected;
var registry = registryClient.GetMain<ServiceRegistry.IClient>();

// Start multiple calculator services
var calc1 = new CalculatorService(registry, 5001);
var calc2 = new CalculatorService(registry, 5002);
var calc3 = new CalculatorService(registry, 5003);

await calc1.StartAsync();
await calc2.StartAsync();
await calc3.StartAsync();

// Create load-balanced client
await using var calcClient = new ServiceClient<Calculator.IClient>(registry, "calculator");
await calcClient.InitializeAsync();

// Make calls (automatically load balanced)
for (int i = 0; i < 10; i++)
{
    var calc = calcClient.GetService();
    var result = await calc.Add(i, i);
    Console.WriteLine($"Call {i}: {i} + {i} = {result}");
}

// Stop services
await calc1.StopAsync();
await calc2.StopAsync();
await calc3.StopAsync();
```

## Features Demonstrated

- **Service registration**: Services register themselves on startup
- **Service discovery**: Clients lookup services by name
- **Load balancing**: Round-robin distribution across instances
- **Dynamic updates**: Clients react to services joining/leaving
- **Health awareness**: Automatic removal of disconnected services
- **Metadata support**: Services can publish version and other info
